    LCPForceFeedback.inl
    MechanicalStateForceFeedback.h
    NullForceFeedback.h
    NullForceFeedbackT.h
    config.h

)

    # EnslavementForceFeedback.cpp
    LCPForceFeedback.cpp
    NullForceFeedback.cpp

    initHaptics.cpp

sofa_install_targets(SofaGeneral ${PROJECT_NAME} ${PROJECT_NAME})
